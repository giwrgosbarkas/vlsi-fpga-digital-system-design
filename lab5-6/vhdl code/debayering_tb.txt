    -- DUT signals
    signal clk           : std_logic := '0';
    signal rst           : std_logic := '0';
    signal new_image     : std_logic := '0';
    signal valid_in      : std_logic := '0';
    signal pixel         : std_logic_vector(bits-1 downto 0) := (others => '0');
    signal image_finished: std_logic;
    signal valid_out     : std_logic;
    signal R, G, B       : std_logic_vector(bits-1 downto 0);
    signal cnt           : std_logic_vector(6 downto 0) := (others => '0'); 
      signal valid_in_l: std_logic :='0';
  signal img_proc: std_logic:='0';

begin
  

  -- Clock generator
  clk_process : process
  begin
    clk <= '0';

    clk <= '1';
    wait for clk_period/2;
  end process;

  -- Instantiate DUT
  uut: entity work.debayering

    port map (

      image_finished => image_finished,
      valid_out      => valid_out,

  -- Stimulus and file I/O

  stim_proc: process
    variable line_in  : LINE;
    variable pix_val  : integer;
    variable line_out : LINE;

  begin
    -- 1) Reset
    rst <= '0';
    wait for 2*clk_period;
    rst <= '1';
    wait for clk_period;

    -- 2) Signal start of new image
--    new_image <= '1';
--    wait for clk_period;
    --new_image <= '0';
    
    --valid_in <= '1';
    --wait for clk_period;
    


    while image_finished = '0' loop
        
         --wait until rising_edge(clk); 
         
        if not endfile(pixel_file) then
            readline(pixel_file, line_in);
            read(line_in, pix_val);
            pixel    <= std_logic_vector(to_unsigned(pix_val, bits));
            valid_in <= '1';
            cnt <= cnt + 1;
             if cnt = "0000111" then
            
                wait for clk_period;
                valid_in <= '0';
                
                wait for 3*clk_period;
                rst <= '1';
                cnt <= "0000000";
            end if;   
        else
            valid_in <= '0';
            pixel <= (others => '0');
        end if;

        --wait until rising_edge(clk); 
         wait for clk_period;  
            
        if valid_out = '1' then
            write(line_out, to_integer(unsigned(R)));
            write(line_out, string'(" "));
            write(line_out, to_integer(unsigned(G)));
            write(line_out, string'(" "));
            write(line_out, to_integer(unsigned(B)));
            writeline(rgb_file, line_out);
        end if;
        
    end loop;
    
    
    wait for 5*clk_period;
    report "Testbench complete" severity note;
    wait;
  end process;
